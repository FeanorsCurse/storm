001    package SAPBAPI;
002    import com.sap.mw.jco.*;
003    
004    public class SAPConnector {
005            JCO.Client mConnection;
006    
007            IRepository mRepository;
008    
009            public SAPConnector() {
010    
011            }
012    
013            public void connectionAttributes() {
014                    System.out.println(mConnection.getAttributes());
015            }
016            
017            public void execute(JCO.Function function){
018                    mConnection.execute(function);
019            }
020    
021            public boolean openConnectionToSAP(String clientSAP, String login,
022                            String passwort, String language, String server, String systemNumber) {
023                    try {
024                            mConnection = JCO.createClient(clientSAP, login, passwort,
025                                            language, server, systemNumber);
026    
027                            mConnection.connect();
028                            return true;
029                    } catch (Exception ex) {
030                            ex.printStackTrace();
031                            return false;
032                    }
033            }
034    
035            public void closeConnectionToSAP() {
036                    mConnection.disconnect();
037            }
038    
039            public JCO.Function createFunction(String name) throws Exception {
040                    try {
041                            mRepository = new JCO.Repository("myRepository", mConnection);
042                            IFunctionTemplate ft = mRepository.getFunctionTemplate(name
043                                            .toUpperCase());
044                            if (ft == null)
045                                    return null;
046                            return ft.getFunction();
047                    } catch (Exception ex) {
048                            throw new Exception(
049                                            "The JCO.Function-Objekt connot be created.");
050                    }
051            }
052    
053    
054    
055    }// Ende der Klasse
